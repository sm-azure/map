152 void RouteExpander<RoutingCostData>::expandSameLaneNeighbors(
153  lane::Lane::ConstPtr lane, typename RouteExpander<RoutingCostData>::RoutingPoint const &origin)
154 {
155  if ((lane->id == getDest().laneId)
156  && ((isPositiveMovement(lane, origin) && (origin.first.point.parametricOffset <= getDest().parametricOffset))
157  || (isNegativeMovement(lane, origin) && (origin.first.point.parametricOffset >= getDest().parametricOffset))))
158  {
159  // approaching the destination from valid side
160  addNeighbor(lane, origin, lane, getRoutingDest(), ExpandReason::Destination);
161  }
162  if (isPositiveMovement(lane, origin) && !isEnd(origin))
163  {
164  addNeighbor(lane,
165  origin,
166  lane,
167  createRoutingParaPoint(lane->id, physics::ParametricValue(1.), origin.first.direction),
168  ExpandReason::SameLaneNeighbor);
169  }
170  if (isNegativeMovement(lane, origin) && !isStart(origin))
171  {
172  addNeighbor(lane,

180 template <class RoutingCostData>
181 void RouteExpander<RoutingCostData>::expandLongitudinalNeighbors(
182  lane::Lane::ConstPtr lane, typename RouteExpander<RoutingCostData>::RoutingPoint const &origin)
183 {
184  lane::ContactLaneList contact_lanes;
185  if (isEnd(origin) && isPositiveMovement(lane, origin))
186  {
187  contact_lanes = getContactLanes(*lane, lane::ContactLocation::SUCCESSOR);
188  }
189  else if (isStart(origin) && isNegativeMovement(lane, origin))
190  {
191  contact_lanes = getContactLanes(*lane, lane::ContactLocation::PREDECESSOR);
192  }
193  for (auto contact_lane : contact_lanes)
194  {
195  lane::Lane::ConstPtr other_lane = lane::getLanePtr(contact_lane.toLane);
196  if (other_lane)
197  {
198  if (isRouteable(*other_lane))
199  {
200  lane::ContactLocation other_to_lane = getContactLocation(*other_lane, lane->id);
201  if (other_to_lane == lane::ContactLocation::SUCCESSOR)
202  {
203  auto routingDirection = RoutingDirection::NEGATIVE;
204  if (origin.first.direction == RoutingDirection::DONT_CARE)
205  {
206  routingDirection = RoutingDirection::DONT_CARE;
207  }
208  addNeighbor(lane,
209  origin,
210  other_lane,
211  createRoutingParaPoint(other_lane->id, physics::ParametricValue(1.), routingDirection),
212  ExpandReason::LongitudinalNeighbor);
213  }
214  else if (other_to_lane == lane::ContactLocation::PREDECESSOR)
215  {
216  auto routingDirection = RoutingDirection::POSITIVE;
217  if (origin.first.direction == RoutingDirection::DONT_CARE)
218  {
219  routingDirection = RoutingDirection::DONT_CARE;
220  }
221  addNeighbor(lane,
222  origin,
223  other_lane,
224  createRoutingParaPoint(other_lane->id, physics::ParametricValue(0.), routingDirection),
225  ExpandReason::LongitudinalNeighbor);
226  }
227  else
228  {
229  throw std::runtime_error("Other lane neither SUCCESSOR not PREDECESSOR!");
230  }
231  }
232  }
233  else
234  {
235  throw std::runtime_error("No other lane!");
236  }
237  }
238 }
