79 PointType getParametricPoint(std::vector<PointType> const &edge,
80  physics::Distance const &edgeLength,
81  const physics::ParametricValue &t)
82 {
83  if (!edge.empty())
84  {
85  physics::Distance length(0);
86  physics::Distance length_t = edgeLength * t;
87  for (size_t i = 0; i < edge.size() - 1; i++)
88  {
89  const auto &pt0 = edge[i];
90  const auto &pt1 = edge[i + 1];
91  auto d = distance(pt0, pt1);
92  if (d != physics::Distance(0.))
93  {
94  auto length_1 = length + d;
95  if (length_1 >= length_t)
96  {
97  auto d_t = length_t - length;
98  physics::ParametricValue tt(d_t / d);
99  return vectorInterpolate(pt0, pt1, tt);
100  }
101  else
102  {
103  length = length_1;
104  }
105  }
106  }
107  return edge.back();
108  }
109  else
110  {
111  return PointType();

213 physics::ParametricValue
214 findNearestPointOnEdge(std::vector<PointType> const &edge, physics::Distance const &edgeLength, const PointType &pt)
215 {
216  if (isValid(pt))
217  {
218  if (edge.size() == 0)
219  {
220  return physics::ParametricValue();
221  }
222  else if ((edge.size() == 1) || (edgeLength == physics::Distance(0.)))
223  {
224  return physics::ParametricValue(0);
225  }
226  else
227  {
228  physics::ParametricValue t_one = findNearestPointOnSegment(pt, edge[0], edge[1]);
229  PointType pt_nearest = vectorInterpolate(edge[0], edge[1], t_one);
230  physics::Distance d_nearest = distance(pt, pt_nearest);
231  physics::Distance offset_nearest = distance(pt_nearest, edge[0]);
232  physics::Distance running_offset = physics::Distance(0);
233  for (size_t i = 1; i + 1 < edge.size(); i++)
234  {
235  physics::ParametricValue t = findNearestPointOnSegment(pt, edge[i], edge[i + 1]);
236  PointType pt_candidate = vectorInterpolate(edge[i], edge[i + 1], t);
237  physics::Distance d = distance(pt_candidate, pt);
238  running_offset += distance(edge[i - 1], edge[i]);
239  if (d < d_nearest)
240  {
241  pt_nearest = pt_candidate;
242  d_nearest = d;
243  offset_nearest = running_offset + distance(pt_nearest, edge[i]);
244  }
245  }
246  return physics::ParametricValue(offset_nearest / edgeLength);
247  }
248  }
249  else
250  {
251  return physics::ParametricValue();
252  }
